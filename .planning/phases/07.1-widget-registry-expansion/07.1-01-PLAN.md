def _register_spinbox(self, parent, *, label, textvariable, from_, to,
                     tab=None, section=None, **kwargs):
    """Create and register a tk.Spinbox (or ttk.Spinbox if requested via
    _use_ttk=True kwarg). Writable; value coerced to float on set."""
    use_ttk = kwargs.pop('_use_ttk', False)
    cls = ttk.Spinbox if use_ttk else tk.Spinbox
    widget = cls(parent, textvariable=textvariable,
                 from_=from_, to=to, **kwargs)
    self._widget_registry_add(label, 'Spinbox', widget, textvariable,
                              tab, section, writable=True)
    return widget

def _register_check(self, parent, *, label, variable, tab=None, section=None, **kwargs):
    """Create and register a tk.Checkbutton. Label is the check's text.
    Writable; value coerced to bool on set."""
    widget = tk.Checkbutton(parent, text=label, variable=variable, **kwargs)
    self._widget_registry_add(label, 'Checkbutton', widget, variable,
                              tab, section, writable=True)
    return widget

def _register_entry(self, parent, *, label, textvariable, tab=None, section=None, **kwargs):
    """Create and register a tk.Entry. Label is explicit (no automatic
    sibling Label detection). Writable; value used verbatim on set."""
    widget = tk.Entry(parent, textvariable=textvariable, **kwargs)
    self._widget_registry_add(label, 'Entry', widget, textvariable,
                              tab, section, writable=True)
    return widget

def _register_listbox(self, parent, *, label, tab=None, section=None, **kwargs):
    """Create and register a tk.Listbox. Writable; value = row index."""
    widget = tk.Listbox(parent, **kwargs)
    self._widget_registry_add(label, 'Listbox', widget, None,
                              tab, section, writable=True)
    return widget

def _register_scale(self, parent, *, label, variable, from_, to,
                   tab=None, section=None, **kwargs):
    """Create and register a tk.Scale (or ttk.Scale if _use_ttk=True).
    Writable; value coerced to float on set."""
    use_ttk = kwargs.pop('_use_ttk', False)
    cls = ttk.Scale if use_ttk else tk.Scale
    widget = cls(parent, variable=variable, from_=from_, to=to, **kwargs)
    self._widget_registry_add(label, 'Scale', widget, variable,
                              tab, section, writable=True)

def _register_notebook(self, widget, *, label='tab'):
    """Register an existing ttk.Notebook (pre-constructed in builders) under

    self._widget_registry_add(label, 'Notebook', widget, None,

**Shared helper `_widget_registry_add`:**

```python
def _widget_registry_add(self, label, wtype, widget, var,
                        tab, section, writable):
    """Single entry point for registry insertion. Thread-safe."""
    entry = {
        'type': wtype,
        'widget': weakref.ref(widget),
        'var': var,
        'tab': tab,
        'section': section,
        'label': label,
        'writable': writable,
    }
    with self._widget_registry_lock:
        # Duplicate labels are tolerated at insertion time; finalize resolves.
        base_id = label
        if base_id in self._widget_registry and tab:
            # Early disambiguation when tab is known
            resolved_id = f'{base_id}@{tab}'
        else:
            resolved_id = base_id
        self._widget_registry[resolved_id] = entry
```

**Finalizer called after `_setup_gui_thread` completes** (D-04 retroactive disambiguation):

```python
def _finalize_widget_registry(self):
    """Scan for label collisions and retroactively add @tab suffixes.
    Called from _setup_gui_thread after all builders complete."""
    with self._widget_registry_lock:
        by_label = {}
        for rid, entry in list(self._widget_registry.items()):
            by_label.setdefault(entry['label'], []).append(rid)
        for label, rids in by_label.items():
            if len(rids) == 1:
                continue
            # Collision — rename each to label@tab (skip entries already @-suffixed)
            for rid in rids:
                if '@' in rid:
                    continue
                entry = self._widget_registry[rid]
                tab = entry.get('tab') or '?'
                new_id = f'{label}@{tab}'
                self._widget_registry[new_id] = entry
                del self._widget_registry[rid]
        self._widget_registry_rebuilding = False
```

Add the three service implementations near the other `_srv_*` methods. Each service:
1. Checks `self._widget_registry_rebuilding` — if True, return `success=false, message='GUI rebuilding — retry in 1s'`.
2. For get/set: reads `widget_id` param; returns error if empty.

**`_srv_list_widgets`:**

Builds and returns a markdown table grouped by tab. Columns:
`| Tab | Section | Label | Type | Current Value | Writable |`

Plus a separate "Agent-only widgets" section for entries with no tab/section.

Current-value retrieval:
- Spinbox/Entry/Scale: read `var.get()` if var is present; else `widget.get()`
- Checkbutton: `'true'` if `var.get()` truthy else `'false'`
- Listbox: format as `'<sel_text> [N/M]'` where N is selected index (-1 if none), M is total
- Notebook: return the active tab text via `notebook.tab(notebook.select(), 'text')`
- Text (logs): summarize as `<N chars, last line: "...">`

**`_srv_get_widget_value`:**

1. Read `widget_id` param.
2. Look up entry; return `success=false, message='widget <id> not found'` if absent.
3. Dispatch by `entry['type']`:
   - Spinbox/Entry/Scale → return `value=<raw_var_get>`
   - Checkbutton → return `value=true|false`
   - Listbox → return `value=<selected_text>, extra=index=N, count=M, items=<pipe-joined>` (D-13, D-14)
   - Notebook → return current tab text
   - Text → return the widget contents bounded to last 10 KB (D-21)
4. Response.message format: `id=<id>, type=<class>, value=<v>[, extra=...]`.
5. Also accepts button IDs — if `widget_id` matches an entry in `self._button_registry`, delegate to `_srv_get_widget_state`'s logic and return that state (D-11).

**`_srv_set_widget_value`:**

1. Read `widget_id` and `widget_value` params.
2. If `widget_id` matches a button, return `success=false, message='buttons are click-only; use ros2 service call ~/<name>'` (D-11).
3. Look up the widget entry; error on miss.
4. Return `success=false, message='widget <id> is read-only'` when `entry['writable']` is False (D-07, D-21).
5. Per-type coercion (D-15..D-18):
   - Spinbox/Scale: parse float. Clamp to registered `from_`/`to` if needed (log warning). If parse fails → error.
   - Checkbutton: parse bool from `{true,false,1,0,yes,no}` case-insensitive.
   - Entry: use verbatim.
   - Listbox: parse as non-negative int; bounds-check.
   - Notebook: match tab text exactly; error listing available tabs.
6. Dispatch the write on the GUI thread via `self.root.after(0, _do_write)` with a `threading.Event` and 1.0s timeout (D-19). Return success/failure based on whether the write completed.
