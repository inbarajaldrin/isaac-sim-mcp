**The demo sequence (write as a shell script `/home/aaugus11/Documents/isaac-sim-mcp/.planning/phases/07.1-widget-registry-expansion/demo_agent_driven_ik.sh`):**

```bash
#!/usr/bin/env bash
# Proves Phase 07.1 SC-6: agent drives full IK flow via CLI alone.
set -euo pipefail
SVC=/so_arm101_control_gui
call() { ros2 service call "${SVC}/$1" std_srvs/srv/Trigger "{}"; }
set_param() { ros2 param set "${SVC}" "$1" "$2"; }
set_widget() { set_param widget_id "$1"; set_param widget_value "$2"; call set_widget_value; }
get_widget() { set_param widget_id "$1"; call get_widget_value; }

# 1. Switch to IK tab
set_widget tab IK

# 2. Type target pose (example coordinates — reachable from home)
set_widget 'X:' 0.20
set_widget 'Y:' 0.00
set_widget 'Z:' 0.15
set_widget 'Roll' 0
set_widget 'shoulder_lift' 90   # pitch — naming matches current RPY inputs
set_widget 'Yaw' 0

# 3. Read back to verify
get_widget 'X:'
get_widget 'Y:'
get_widget 'Z:'

# 4. Plan & Execute via the IK Trigger service
call ik_plan_execute

# 5. Read the process log to verify IK succeeded
sleep 2
get_widget process_log

# 6. Capture a screenshot for the record
set_param ik_target '/tmp/07.1-demo-after-ik.png'
call screenshot
```
