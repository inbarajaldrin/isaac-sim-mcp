```python
#!/usr/bin/env python3
"""probe_unified_usd.py — enumerate USD prim names under the unified robot USD and
diff against the live aic_eval TF frame list (aic_frames_live.gv).

Usage:
    ~/env_isaaclab/bin/python exts/aic-dt/scripts/probe_unified_usd.py \\
        [USD_PATH] [FRAMES_GV_PATH]

Defaults:
    USD_PATH        = exts/aic-dt/assets/robot/aic_unified_robot_cable_sdf.usd
    FRAMES_GV_PATH  = .planning/phases/01-foundation-parity/aic_frames_live.gv

Output (stdout, also captured to usd_prim_inventory.txt by the caller):
    [USD prims under /World or /World/UR5e]
    <list of all Xform prim paths>

    [Live frame names from aic_frames_live.gv]
    <unique sorted frame names>

    [Diff: USD-prim-name → expected-ros-frame-id]
    <table with one row per live frame: prim_match | prim_path | expected_frame_id | match_status>

    [Recommended fix strategy]
    <strategy name + rationale>
"""
import os, re, sys
from pathlib import Path
from pxr import Usd

DEFAULT_USD = "exts/aic-dt/assets/robot/aic_unified_robot_cable_sdf.usd"
DEFAULT_GV  = ".planning/phases/01-foundation-parity/aic_frames_live.gv"

def parse_frames(gv_path):
    text = Path(gv_path).read_text()
    return sorted(set(re.findall(r'"([^"]+)"', text)))

def enumerate_prims(usd_path):
    stage = Usd.Stage.Open(usd_path)
    if not stage:
        raise RuntimeError(f"Cannot open USD: {usd_path}")
    prims = []
    for prim in stage.Traverse():
        if prim.IsA(Usd.Prim) and (prim.GetTypeName() in ("Xform", "PhysicsRevoluteJoint", "PhysicsPrismaticJoint", "PhysicsFixedJoint", "Joint")):
            prims.append(str(prim.GetPath()))
    return sorted(prims)

def underscore_to_slash_candidates(name):
    """For a USD-style name like gripper_hande_base_link, candidates are:
    gripper/hande_base_link, gripper_hande_base_link, gripper/hande/base_link.
    Return all reasonable mappings to compare against ROS frame names."""
    parts = name.split("_")
    cands = {name}
    # try one-slash insertions
    for i in range(1, len(parts)):
        cands.add("/".join(["_".join(parts[:i]), "_".join(parts[i:])]))
    return cands

def main():
    usd_path = sys.argv[1] if len(sys.argv) > 1 else DEFAULT_USD
    gv_path  = sys.argv[2] if len(sys.argv) > 2 else DEFAULT_GV
    print(f"USD: {usd_path}")
    print(f"GV : {gv_path}\n")

    frames = parse_frames(gv_path)
    prims  = enumerate_prims(usd_path)

    print("[USD prims under stage]")
    for p in prims:
        print(f"  {p}")

    print("\n[Live frame names from aic_frames_live.gv]")
    for f in frames:
        print(f"  {f}")

    print("\n[Diff: live frame → USD prim match]")
    mismatches = []
    for f in frames:
        # extract last path segment for comparison
        f_us = f.replace("/", "_")
        match = None
        for p in prims:
            leaf = p.rsplit("/", 1)[-1]
            if leaf == f or leaf == f_us:
                match = p
                break
        status = "MATCH" if match and match.rsplit("/",1)[-1] == f else ("UNDERSCORE-MISMATCH" if match else "NO-PRIM")
        print(f"  {status:22s}  live='{f}'  prim='{match or '<none>'}'")
        if status != "MATCH":
            mismatches.append((f, match))

    print(f"\n[Summary] {len(frames)} live frames | {len([m for m in mismatches if m[1] is not None])} underscore-mismatches | {len([m for m in mismatches if m[1] is None])} no-prim")

    print("\n[Recommended fix strategy]")
    if not mismatches:
        print("  STRATEGY: NONE — prim names already match live frame_ids; default ROS2PublishTransformTree mapping is correct.")
    elif len([m for m in mismatches if m[1] is None]) > 0:
        print("  STRATEGY: REVIEW — some live frames have no matching USD prim. Investigate before choosing sublayer-rename vs Raw-publisher-override.")
    else:
        # Underscore-mismatches only → both strategies viable; recommend sublayer if all are underscore→slash
        slashed_only = all("/" in f for f, _ in mismatches)
        if slashed_only:
            print(f"  STRATEGY: USD-SUBLAYER-RENAME — {len(mismatches)} prims need underscore-to-slash renaming. Author exts/aic-dt/assets/robot/aic_unified_robot_frame_overrides.usda with prim-name overrides.")
        else:
            print(f"  STRATEGY: PER-FRAME-RAW-OVERRIDE — heterogeneous mismatch types; use ROS2PublishRawTransformTree for {len(mismatches)} frames listed above. Plan 06 TF builder reads this list.")

    # Emit machine-parseable override list for Plan 06 (or sublayer authoring)
    print("\n[Override list for Plan 06 / sublayer]")
    for live_frame, prim_path in mismatches:
        print(f"  OVERRIDE  prim='{prim_path}'  expected_frame_id='{live_frame}'")

if __name__ == "__main__":
    main()
```

#!/usr/bin/env bash
# probe_aic_controller_jointstate.sh — inspect aic_controller's /joint_states
# subscriber to determine name-indexed vs positional access (D-11 / RESEARCH Pitfall #1).
#
# Usage: ./probe_aic_controller_jointstate.sh [AIC_CONTROLLER_SRC_DIR]
# Default: ~/Documents/aic/aic_controller/src

set -euo pipefail
SRC="${1:-$HOME/Documents/aic/aic_controller/src}"

echo "Probing: $SRC"
if [ ! -d "$SRC" ]; then
  echo "ERROR: Source dir not found. Either pass the path as arg 1 or check out ~/Documents/aic."
  echo "Per D-13 the runtime works without ~/Documents/aic, but THIS PROBE NEEDS THE AIC REPO."
  exit 2
fi

echo
echo "[1] Searching for JointState subscriber declarations..."
grep -rn "JointState" "$SRC" --include='*.cpp' --include='*.hpp' --include='*.h' --include='*.cc' || \
  echo "  (no JointState references found in $SRC)"

echo
echo "[2] Searching for joint_states subscription topic..."
grep -rn 'joint_states' "$SRC" --include='*.cpp' --include='*.hpp' --include='*.h' --include='*.cc' || \
  echo "  (no joint_states topic literal found)"

echo
echo "[3] Searching for name-indexed access patterns (msg->name, name[i])..."
NAME_ACCESS=$(grep -rEn '(msg->name|->name\[|\.name\[|find\(.*joint|index\(.*joint)' "$SRC" --include='*.cpp' --include='*.hpp' --include='*.h' --include='*.cc' | grep -i 'joint\|state' || true)
if [ -n "$NAME_ACCESS" ]; then
  echo "$NAME_ACCESS"
  echo "  → name-indexed access PRESENT"
else
  echo "  → no name-indexed access patterns found"
fi

echo
echo "[4] Searching for positional access patterns (msg->position[N], direct index)..."
POS_ACCESS=$(grep -rEn 'position\[[0-9]+\]|->position\[|\.position\[' "$SRC" --include='*.cpp' --include='*.hpp' --include='*.h' --include='*.cc' || true)
if [ -n "$POS_ACCESS" ]; then
  echo "$POS_ACCESS"
  echo "  → positional access PRESENT"
else
  echo "  → no positional access patterns found"
fi

echo
echo "[5] Verdict (heuristic):"
if [ -n "$NAME_ACCESS" ] && [ -z "$POS_ACCESS" ]; then
  echo "  CONCLUSION: NAME-INDEXED — joint name ordering does NOT functionally matter."
  echo "  Implication: Plan 06's ROS2PublishJointState with articulation-discovery order is sufficient."
elif [ -z "$NAME_ACCESS" ] && [ -n "$POS_ACCESS" ]; then
  echo "  CONCLUSION: POSITIONAL — joint name ordering IS functionally required."
  echo "  Implication: Plan 05 Task 4 (reordering wrapper) MUST be added."
elif [ -n "$NAME_ACCESS" ] && [ -n "$POS_ACCESS" ]; then
  echo "  CONCLUSION: MIXED — both patterns present; manually review before deciding. Default-safe: POSITIONAL (add wrapper)."
else
  echo "  CONCLUSION: UNKNOWN — neither pattern found in C++ source. Manually inspect the controller plugin."
fi

**Part A: Author `exts/aic-dt/assets/robot/aic_unified_robot_frame_overrides.usda`.**

USD sublayers compose into the base layer. The override sublayer reparents (via `Sdf.PrimSpec`-style overrides) the prims whose ROS frame_ids must contain `/`. USD does NOT allow `/` in prim names directly, but it DOES allow `:` as a namespace separator in some contexts. **However, ROS2PublishTransformTree's prim-name → frame_id derivation is documented as a literal copy of the prim's name string.** The viable approach is:

1. Author the sublayer with `over` blocks that define a `customData` field `frameId` on each override-target prim. (If `ROS2PublishTransformTree` does NOT honor `frameId` customData — which RESEARCH.md Pitfall #3 strongly suggests — fall back to the `PER-FRAME-RAW-OVERRIDE` strategy. **Test this empirically in Plan 06 verify; this plan's sublayer is a hypothesis that may be invalidated.**)
2. Alternatively, the sublayer can rename child prims to use `:` namespace separators, which `ROS2PublishTransformTree` may flatten to `/` (some Isaac Sim versions do this; verify).

For each override entry from `usd_prim_inventory.txt`, emit an `over` block:

```usda
#usda 1.0
(
    """Frame-name override sublayer for AIC topic-parity (Phase 1 D-10 / PARITY-04).

    Authored by Plan 05 Task 3. Read by extension.py:load_robot via
    Sdf.Layer.AddSubLayerPath() so ROS2PublishTransformTree produces frame_ids
    matching the live aic_eval TF tree (gripper/hande_base_link, etc.).

    If empirical verification (Plan 06 verify) shows the customData:frameId
    approach does NOT propagate to ROS2PublishTransformTree, this sublayer
    is replaced by per-frame ROS2PublishRawTransformTree nodes in Plan 06's
    TF builder.
    """
)

over "World"
{
    over "UR5e"
    {
        over "aic_unified_robot"
        {
            over "gripper_hande_base_link"
            {
                custom string customData:frameId = "gripper/hande_base_link"
            }
            # ... one over block per override entry
        }
    }
}
```

The exact `over` structure depends on the captured prim paths. Author one `over` block per override entry from `usd_prim_inventory.txt`'s `[Override list for Plan 06 / sublayer]` section.

```python
def setup_joint_state_reorder(self):
    """Phase-1 stop-gap reordering bridge for /joint_states (PARITY-03 / D-11).

    aic_controller's subscriber accesses positionally per Plan 05 Task 2 probe.
    This bridge subscribes to /joint_states_isaac_raw (Isaac Sim's articulation-
    discovery order) and republishes on /joint_states in alphabetical order
    matching live aic_eval.

    TODO(Phase 2): Replace with pure OmniGraph SubscribeJointState→PublishJointState
    chain to honor the no-bridge architectural law.
    """
    import rclpy
    from rclpy.node import Node
    from sensor_msgs.msg import JointState
    import threading

    if getattr(self, '_joint_reorder_node', None) is not None:
        return  # Already running

    if not rclpy.ok():
        rclpy.init(args=None)

    class JointReorderNode(Node):
        def __init__(self, target_order):
            super().__init__('aic_dt_joint_state_reorder')
            self._target_order = target_order
            self._pub = self.create_publisher(JointState, '/joint_states', 10)
            self._sub = self.create_subscription(JointState, '/joint_states_isaac_raw', self._cb, 10)

        def _cb(self, msg):
            try:
                # Build name → (position, velocity, effort) lookup
                idx = {n: i for i, n in enumerate(msg.name)}
                out = JointState()
                out.header = msg.header
                out.name = list(self._target_order)
                out.position = [msg.position[idx[n]] for n in self._target_order if n in idx]
                if msg.velocity:
                    out.velocity = [msg.velocity[idx[n]] for n in self._target_order if n in idx]
                if msg.effort:
                    out.effort   = [msg.effort[idx[n]]   for n in self._target_order if n in idx]
                self._pub.publish(out)
            except Exception as e:
                self.get_logger().warn(f'reorder skipped: {e}')

    self._joint_reorder_node = JointReorderNode(AIC_JOINT_NAMES_ALPHABETICAL)
    self._joint_reorder_thread = threading.Thread(
        target=lambda: rclpy.spin(self._joint_reorder_node), daemon=True)
    self._joint_reorder_thread.start()
    print("[AIC-DT] JointState reorder bridge started: /joint_states_isaac_raw → /joint_states (alphabetical, 7 joints)")
```
