```python
from pxr import Usd, UsdGeom, Sdf
import os, sys

def author_mount_usd(folder: str, usd_name: str):
    usd_path = os.path.join(folder, usd_name)
    stage = Usd.Stage.CreateNew(usd_path)
    root = UsdGeom.Xform.Define(stage, "/Root")
    # Find the mesh source (prefer .dae if it has materials baked, else .stl)
    mesh_files = sorted([f for f in os.listdir(folder) if f.lower().endswith((".dae", ".stl", ".obj"))])
    if not mesh_files:
        raise RuntimeError(f"No mesh found in {folder}")
    mesh_src = mesh_files[0]
    # Use Sdf payload reference so the mesh is loaded lazily by Isaac Sim's USD resolver
    mesh_prim = stage.DefinePrim("/Root/Mesh", "Xform")
    mesh_prim.GetReferences().AddReference(f"./{mesh_src}")
    stage.SetDefaultPrim(root.GetPrim())
    stage.GetRootLayer().Save()
    print(f"[build_mount_rail_usds] wrote {usd_path} referencing {mesh_src}")

if __name__ == "__main__":
    base = sys.argv[1] if len(sys.argv) > 1 else "exts/aic-dt/assets/assets"
    for name, usd in [
        ("LC Mount",  "lc_mount_visual.usd"),
        ("SFP Mount", "sfp_mount_visual.usd"),
        ("SC Mount",  "sc_mount_visual.usd"),
    ]:
        folder = os.path.join(base, name)
        if os.path.isdir(folder):
            try:
                author_mount_usd(folder, usd)
            except Exception as e:
                print(f"[build_mount_rail_usds] WARN {name}: {e}")
                # Fall through — Task 2 atoms will degrade gracefully
```

```python
def _cmd_spawn_task_board_base(self, x: float = 0.25, y: float = 0.0, z: float = 1.14,
                               roll: float = 0.0, pitch: float = 0.0, yaw: float = 0.0) -> Dict[str, Any]:
    """SCENE-01: Spawn task board base. Parameter names mirror spawn_task_board.launch.py."""
    try:
        prim_path = "/World/TaskBoard"
        usd_rel = AIC_OBJECTS["task_board_base"]["usd"]  # "assets/Task Board Base/base_visual.usd" per Plan 04
        usd_uri = self._local_asset(usd_rel)
        # Idempotent cleanup
        from pxr import Usd, UsdGeom, Gf, Sdf
        stage = omni.usd.get_context().get_stage()
        if stage.GetPrimAtPath(prim_path).IsValid():
            stage.RemovePrim(prim_path)
        # Author Xform + reference USD
        xform = UsdGeom.Xform.Define(stage, prim_path)
        xform.GetPrim().GetReferences().AddReference(usd_uri)
        # Apply pose: translate then rotate (XYZ Euler intrinsic, matching spawn_task_board.launch.py's -R -P -Y semantics)
        xform.AddTranslateOp().Set(Gf.Vec3d(float(x), float(y), float(z)))
        xform.AddRotateXYZOp().Set(Gf.Vec3f(float(roll), float(pitch), float(yaw)))
        return {"status": "success", "message": f"Task board base spawned at {prim_path} pose=({x},{y},{z}, RPY={roll},{pitch},{yaw})"}
    except Exception as e:
        return {"status": "error", "message": f"spawn_task_board_base failed: {e}"}
```

```python
def _quat_to_rpy(self, quat):
    if quat is None:
        return (0.0, 0.0, 0.0)
    w, x, y, z = quat
    # ZYX intrinsic → roll (X), pitch (Y), yaw (Z)
    import math
    sinr_cosp = 2.0 * (w * x + y * z)
    cosr_cosp = 1.0 - 2.0 * (x * x + y * y)
    roll = math.atan2(sinr_cosp, cosr_cosp)
    sinp = 2.0 * (w * y - z * x)
    pitch = math.asin(max(-1.0, min(1.0, sinp)))
    siny_cosp = 2.0 * (w * z + x * y)
    cosy_cosp = 1.0 - 2.0 * (y * y + z * z)
    yaw = math.atan2(siny_cosp, cosy_cosp)
    return (roll, pitch, yaw)
```
