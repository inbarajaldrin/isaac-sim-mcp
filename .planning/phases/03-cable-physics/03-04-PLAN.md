### Task 3 — New /scoring/tf publisher

```python
# In start():
scoring_tf_qos = QoSProfile(reliability=RELIABLE, history=KEEP_LAST, depth=10)
self._scoring_tf_pub = self._node.create_publisher(TFMessage, "/scoring/tf", scoring_tf_qos)

# In _on_physics_step (after the lazy articulation init guard):
self._publish_scoring_tf(now)

# New method:
def _publish_scoring_tf(self, stamp):
    """PARITY-08: TFMessage of cable link transforms only."""
    msg = TFMessage()
    cable = self._stage.GetPrimAtPath("/World/UR5e/cable")
    if not cable or not cable.IsValid() or not cable.IsActive():
        return  # cable subtree not active (SCENE-05 not yet enabled)
    for prim in Usd.PrimRange(cable):
        if not UsdPhysics.RigidBodyAPI(prim):
            continue
        xform = self._world_xform(prim)
        if xform is None:
            continue
        t = self._build_transform_msg("world", str(prim.GetName()), stamp, xform)
        msg.transforms.append(t)
    self._scoring_tf_pub.publish(msg)
```
