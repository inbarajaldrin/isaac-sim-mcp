```python
def _setup_insertion_contact_subscription(self):
    """Subscribe to omni.physx contact-report events; filter for plug↔port contacts."""
    from omni.physx import get_physx_simulation_interface
    self._insertion_contact_sub = (
        get_physx_simulation_interface()
        .subscribe_contact_report_events(self._on_insertion_contact_event)
    )
    # Apply PhysxContactReportAPI(threshold=0.0) on the plug-end link prim
    # so contacts involving it are reported. (RigidBodyAPI guard per
    # robot-collision-forensics skill).
    plug_link = self._stage.GetPrimAtPath(self._plug_end_link_path)
    if plug_link and plug_link.IsValid() and plug_link.HasAPI(UsdPhysics.RigidBodyAPI):
        contact_api = PhysxSchema.PhysxContactReportAPI.Apply(plug_link)
        contact_api.CreateThresholdAttr(0.0)
```

`_plug_end_link_path` set during start() — derived from cable_type via Plan 03-03's plug-link discovery helper.

### Task 4 — Contact callback + sustained-contact gate

```python
def _on_insertion_contact_event(self, contact_headers, contact_data):
    """Called from physics thread. O(1) append to a deque; per-tick drain
    in _on_physics_step decides whether to publish."""
    for header in contact_headers:
        actor0 = str(PhysicsSchemaTools.intToSdfPath(header.actor0))
        actor1 = str(PhysicsSchemaTools.intToSdfPath(header.actor1))
        if not self._is_plug_port_pair(actor0, actor1):
            continue
        self._INSERTION_CONTACT_EVENTS.append(("contact", time.time()))

def _is_plug_port_pair(self, a, b):
    """True if (a,b) match (plug-end link, port link) in either order."""
    plug = self._plug_end_link_path
    port = self._port_link_path
    return ({a, b} == {plug, port}) or (a.startswith(plug) and b.startswith(port)) \
           or (a.startswith(port) and b.startswith(plug))

def _maybe_publish_insertion_event(self):
    """Per-tick drain: count contacts in last 5 ticks (~80ms). If sustained
    AND not already armed, publish event + arm. Re-arms when contacts cease."""
    # ... sustained-contact + edge-detection logic
    if self._insertion_event_armed:
        return  # don't spam
    sustained = sum(1 for evt in self._INSERTION_CONTACT_EVENTS
                    if time.time() - evt[1] < 0.1) >= 5
    if sustained:
        msg = String()
        msg.data = self._INSERTION_EVENT_PAYLOAD  # from Plan 03-01 snapshot
        self._scoring_event_pub.publish(msg)
        self._insertion_event_armed = True
    elif not any(time.time() - evt[1] < 0.1 for evt in self._INSERTION_CONTACT_EVENTS):
        # Contacts ceased — re-arm
        self._insertion_event_armed = False
```

Call `_maybe_publish_insertion_event()` from `_on_physics_step` after the existing publish chain.
