       ```python
       def _drop_callback(self, msg):
           """Populate _drop_data from /drop_poses TFMessage."""
           with self._drop_lock:
               for tf in msg.transforms:
                   name = tf.child_frame_id
                   self._drop_data[name] = {
                       'x': tf.transform.translation.x,
                       'y': tf.transform.translation.y,
                       'z': tf.transform.translation.z,
                       'qx': tf.transform.rotation.x,
                       'qy': tf.transform.rotation.y,
                       'qz': tf.transform.rotation.z,
                       'qw': tf.transform.rotation.w,
                   }

       def _update_drop_topic(self, topic='/drop_poses'):
           """Switch drop subscription to topic."""
           if self._drop_sub is not None:
               self.destroy_subscription(self._drop_sub)
           with self._drop_lock:
               self._drop_data.clear()
           self._drop_sub = self.create_subscription(
               TFMessage, topic, self._drop_callback, 10)
           self._append_log(f'Drop topic: {topic}')

       def _populate_drop_list(self):
           """Refresh drop_listbox from _drop_data (call on GUI thread)."""
           if not hasattr(self, '_drop_listbox'):
               return
           self._drop_listbox.delete(0, tk.END)
           with self._drop_lock:
               for name, pos in self._drop_data.items():
                   self._drop_listbox.insert(
                       tk.END,
                       f'{name}  ({pos["x"]:.3f}, {pos["y"]:.3f}, {pos["z"]:.3f})')
           count = self._drop_listbox.size()
           if count > 0:
               self._append_log(f'Drop targets refreshed: {count} found')

       def _get_selected_drop_pose(self):
           """Return (name, x, y, z) for the selected drop listbox entry, or None."""
           if not hasattr(self, '_drop_listbox'):
               return None
           sel = self._drop_listbox.curselection()
           if not sel:
               self._append_log('No drop target selected', 'warn')
               return None
           text = self._drop_listbox.get(sel[0])
           name = text.split('  ')[0]
           with self._drop_lock:
               pose = self._drop_data.get(name)
           if pose is None:
               self._append_log(f'Drop target {name!r} not in data', 'warn')
               return None
           return name, pose['x'], pose['y'], pose['z']
       ```

    ```python
    # ------------------------------------------------------------------
    # Drop motion commands (auto-registered as ~/drop_point, ~/drop_sweep,
    # ~/drop_release Trigger services via the _cmd_* convention)
    # ------------------------------------------------------------------

    def _cmd_drop_point(self):
        """Rotate shoulder_pan to face the selected drop target. ARM-02, ARM-06."""
        result = self._get_selected_drop_pose()
        if result is None:
            return
        name, x, y, z = result

        # ARM-06: pre-validate workspace bounds before any motion
        ok, reason = check_grasp_reachable(x, y, z)
        if not ok:
            self._append_log(
                f'Drop rejected (Point): {reason} ({x:.3f}, {y:.3f}, {z:.3f})', 'warn')
            return

        def _run():
            from so_arm101_control.compute_workspace import geometric_ik, X_PAN
            pan = math.atan2(-y, x - X_PAN)
            with self.joint_lock:
                current = dict(self.joint_positions)
            target = dict(current)
            target['shoulder_pan'] = pan
            self._append_log(
                f'Drop Point: pan={math.degrees(pan):.1f}° toward {name}')
            self._send_arm_goal(target, duration_s=1.0)

        threading.Thread(target=_run, daemon=True).start()

    def _cmd_drop_sweep(self):
        """Sweep wrist_flex from 90° to 0° to place object over cup. ARM-03, ARM-06."""
        result = self._get_selected_drop_pose()
        if result is None:
            return
        name, x, y, z = result

        # ARM-06: pre-validate workspace bounds
        ok, reason = check_grasp_reachable(x, y, z)
        if not ok:
            self._append_log(
                f'Drop rejected (Sweep): {reason} ({x:.3f}, {y:.3f}, {z:.3f})', 'warn')
            return

        duration = self._drop_duration_var.get()

        def _run():
            from so_arm101_control.compute_workspace import geometric_ik, X_PAN
            # Compute IK at target with wrist_flex=0 to get shoulder_lift + elbow_flex
            solutions = geometric_ik(x, y, z, grasp_yaw=math.atan2(-y, x - X_PAN))
            if not solutions:
                self._append_log(
                    f'Drop Sweep: no IK solution for {name} '
                    f'({x:.3f}, {y:.3f}, {z:.3f})', 'warn')
                return

            sol = solutions[0]  # elbow-up preferred
            with self.joint_lock:
                current = dict(self.joint_positions)

            # Hold pan/lift/elbow from IK solution; sweep wrist_flex 90°→0°
            # Build multi-point trajectory: start at current wrist_flex, end at 0
            start_wf = current.get('wrist_flex', math.pi / 2)
            mid = dict(sol)
            mid['wrist_flex'] = start_wf   # hold start angle at t=0

            end = dict(sol)
            end['wrist_flex'] = 0.0        # final angle at t=duration

            self._append_log(
                f'Drop Sweep: wrist_flex {math.degrees(start_wf):.1f}°→0° '
                f'over {duration:.1f}s toward {name}')
            # Animate via execute_trajectory (interpolates from start to end)
            self._execute_trajectory(end, duration_s=duration)

        threading.Thread(target=_run, daemon=True).start()

    def _cmd_drop_release(self):
        """Open gripper to release held object into cup. ARM-04."""
        open_pos = JOINT_LIMITS['gripper_joint'][1]   # max = open
        self._append_log(
            f'Drop Release: opening gripper to {math.degrees(open_pos):.1f}°')
        self._send_gripper_goal(open_pos, duration_s=1.0)
    ```
