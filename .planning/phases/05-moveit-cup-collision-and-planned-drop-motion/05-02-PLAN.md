```python
def _cmd_drop_sweep(self):
    """MoveIt-planned drop: navigates over cup edge with collision avoidance.

    Locks shoulder_pan (set by drop_point) and wrist_roll. Lets MoveIt plan
    shoulder_lift, elbow_flex, wrist_flex to reach drop position above cup
    while avoiding cup collision geometry.
    """
    result = self._get_selected_drop_pose()
    if result is None:
        return
    name, x, y, z = result

    if not MOVEIT_AVAILABLE or not hasattr(self, 'plan_client') or self.plan_client is None:
        self._append_log('MoveIt not available — cannot plan drop sweep', 'warn')
        return
    if not self.plan_client.service_is_ready():
        self._append_log('/plan_kinematic_path not ready', 'warn')
        return

    duration = self._drop_duration_var.get()
    evt = threading.Event()
    self._motion_event = evt

    with self.joint_lock:
        current = dict(self.joint_positions)

    from so_arm101_control.compute_workspace import WRIST_ROLL_URDF_PITCH

    # Goal: wrist forward, roll aligned, pan locked, lift/elbow free to vary
    goal = dict(current)
    goal['wrist_flex'] = 0.0
    goal['wrist_roll'] = -(math.pi / 2 + WRIST_ROLL_URDF_PITCH)
    # shoulder_pan stays at current (set by drop_point)
    # shoulder_lift and elbow_flex stay at current — MoveIt can adjust

    # Tight tolerance = locked, loose tolerance = MoveIt can vary
    tolerances = {
        'shoulder_pan':   0.001,  # LOCKED (set by drop_point)
        'shoulder_lift':  0.05,   # FREE — MoveIt varies for collision avoidance
        'elbow_flex':     0.05,   # FREE — MoveIt varies for collision avoidance
        'wrist_flex':     0.01,   # tight — must reach horizontal
        'wrist_roll':     0.001,  # LOCKED
    }

    constraints = Constraints()
    for jname in ARM_JOINT_NAMES:
        jc = JointConstraint()
        jc.joint_name = jname
        jc.position = goal[jname]
        tol = tolerances[jname]
        jc.tolerance_above = tol
        jc.tolerance_below = tol
        jc.weight = 1.0
        constraints.joint_constraints.append(jc)

    request = GetMotionPlan.Request()
    mpr = MotionPlanRequest()
    mpr.group_name = 'arm'
    mpr.num_planning_attempts = 20
    mpr.allowed_planning_time = 15.0
    vel_scale = self.velocity_scale_var.get()
    mpr.max_velocity_scaling_factor = vel_scale
    mpr.max_acceleration_scaling_factor = vel_scale
    mpr.goal_constraints.append(constraints)
    request.motion_plan_request = mpr

    self._append_log(
        f'Drop Sweep (MoveIt): planning to wrist_flex=0° wrist_roll='
        f'{math.degrees(goal["wrist_roll"]):.1f}° toward {name}')

    def _plan_and_execute():
        future = self.plan_client.call_async(request)
        self._wait_future(future, timeout_sec=20.0)
        resp = future.result()
        if resp is None:
            self._append_log('Drop sweep planning timed out', 'warn')
            evt.set()
            return

        error_code = resp.motion_plan_response.error_code.val
        if error_code != 1:
            self._append_log(
                f'Drop sweep planning failed (error {error_code}). '
                f'Try repositioning arm or adjusting cup position.', 'warn')
            evt.set()
            return

        traj = resp.motion_plan_response.trajectory
        n_pts = len(traj.joint_trajectory.points)
        pt = resp.motion_plan_response.planning_time
        self._append_log(f'Drop sweep plan found ({pt:.3f}s, {n_pts} pts)')

        # Display in RViz
        from moveit_msgs.msg import DisplayTrajectory
        display_msg = DisplayTrajectory()
        display_msg.trajectory.append(traj)
        self._display_traj_pub.publish(display_msg)

        # Extract final point as target for _execute_trajectory
        final_pt = traj.joint_trajectory.points[-1]
        joint_names = traj.joint_trajectory.joint_names
        target = {}
        for i, jn in enumerate(joint_names):
            target[jn] = final_pt.positions[i]

        self._append_log(
            f'Drop Sweep executing: lift={math.degrees(target.get("shoulder_lift", 0)):.1f}° '
            f'elbow={math.degrees(target.get("elbow_flex", 0)):.1f}° '
            f'wrist_flex={math.degrees(target.get("wrist_flex", 0)):.1f}°')
        self._execute_trajectory(target, duration_s=duration,
                                 on_complete=evt.set)

    threading.Thread(target=_plan_and_execute, daemon=True).start()
```

IMPORTANT considerations:
- Do NOT send the full MoveIt trajectory to arm_controller (the action server may not accept multi-point trajectories from MoveIt). Instead, extract the FINAL joint positions and use _execute_trajectory (single-point) like all other commands. The collision avoidance is in the planning — if the goal is reachable without collision, the straight-line interpolation _execute_trajectory does will also be collision-free for this simple 1-joint-dominant motion.
- If this proves insufficient (intermediate collision), a follow-up can send the full multi-point trajectory. But start with the simpler approach.
- The _motion_event / evt.set() pattern MUST be maintained — drop_sweep is called by debug services that block until complete.
