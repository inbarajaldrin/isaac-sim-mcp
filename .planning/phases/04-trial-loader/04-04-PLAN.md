### Subscriber subprocess (the per-trial outcome capture)

To switch ROS_DOMAIN_ID per sim, **fork a subscriber Python subprocess** with the appropriate env. Inline the subscriber as a heredoc'd script within parity_report.py:

```python
SUBSCRIBER_SCRIPT = r'''
# Reference: 04-RESEARCH.md Q3 — bounded rclpy subscriber pattern
import rclpy, sys, json, time, signal
from rclpy.qos import QoSProfile, ReliabilityPolicy, HistoryPolicy
from std_msgs.msg import String

def run(timeout_s, label):
    rclpy.init()
    node = rclpy.create_node(f"parity_report_{label}")
    state = {"insertion_event_fired": False, "off_limit_count": 0,
             "first_event_at": None, "start_at": time.time(), "last_payload": None}

    def on_insert(msg):
        if not state["insertion_event_fired"]:
            state["first_event_at"] = time.time()
            state["insertion_event_fired"] = True
        state["last_payload"] = msg.data

    def on_off_limit(msg):
        n = len(getattr(msg, "contacts", []) or [])
        if n > 0:
            state["off_limit_count"] += n

    qos = QoSProfile(depth=10, reliability=ReliabilityPolicy.BEST_EFFORT,
                     history=HistoryPolicy.KEEP_LAST)
    node.create_subscription(String, "/scoring/insertion_event", on_insert, qos)
    try:
        from ros_gz_interfaces.msg import Contacts
        node.create_subscription(Contacts, "/aic/gazebo/contacts/off_limit", on_off_limit, qos)
    except ImportError:
        pass  # ros_gz_interfaces only on Isaac Sim host humble env

    deadline = time.time() + timeout_s
    while time.time() < deadline and not state["insertion_event_fired"]:
        rclpy.spin_once(node, timeout_sec=0.1)
    # After insertion_event fires, sample for 1s more to count off_limit cleanup
    extra_deadline = time.time() + 1.0
    while time.time() < extra_deadline:
        rclpy.spin_once(node, timeout_sec=0.05)

    duration = (state.get("first_event_at") or time.time()) - state["start_at"]
    rclpy.shutdown()
    return {"insertion_event_fired": state["insertion_event_fired"],
            "off_limit_count": state["off_limit_count"],
            "duration_s": round(duration, 2),
            "last_payload": state["last_payload"]}

if __name__ == "__main__":
    timeout_s = float(sys.argv[1])
    label = sys.argv[2]
    out = run(timeout_s, label)
    print(json.dumps(out))
'''
```

Helper to write the subscriber to a temp file and invoke:
```python
def capture_outcome(domain_id: int, timeout_s: float, label: str) -> dict:
    """Fork a subscriber process with ROS_DOMAIN_ID override; capture outcome JSON."""
    import tempfile
    with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
        f.write(SUBSCRIBER_SCRIPT)
        script_path = f.name
    env = os.environ.copy()
    env["ROS_DOMAIN_ID"] = str(domain_id)
    env.setdefault("RMW_IMPLEMENTATION", "rmw_fastrtps_cpp")
    py = os.path.expanduser("~/env_isaaclab/bin/python")
    try:
        result = subprocess.run([py, script_path, str(timeout_s), label],
                                env=env, capture_output=True, text=True,
                                timeout=timeout_s + 10)
        if result.returncode != 0:
            return {"error": result.stderr.strip()[-500:],
                    "insertion_event_fired": False, "off_limit_count": 0,
                    "duration_s": timeout_s, "last_payload": None}
        return json.loads(result.stdout.strip().splitlines()[-1])
    finally:
        os.unlink(script_path)
```

    """Drive trial_key against Isaac Sim via the 04-03 wrapper. Capture outcome
    on ROS_DOMAIN_ID=7 in parallel."""

        outcome_holder["v"] = capture_outcome(ISAAC_DOMAIN_ID, time_limit_s + SUBSCRIBER_TIMEOUT_BUDGET, f"isaac_{trial_key}")
    t = threading.Thread(target=_capture, daemon=True)

def render_markdown(rows: list, mismatches: int) -> str:
    md = ["# Parity Report — sample_config.yaml × {Isaac Sim, Gazebo}",
          f"\n**Generated:** {time.strftime('%Y-%m-%dT%H:%M:%S')}",
          f"\n**Source config:** {DEFAULT_CONFIG}",
          "\n| Trial | Sim | insertion_event | off_limit | duration_s | pass/fail |",
          "|---|---|---|---|---|---|"]
    for r in rows:
        ev = "✓" if r.get("insertion_event_fired") else "✗"
        flag = " 🔴" if r.get("_mismatch") else ""
        md.append(f"| {r['trial_key']} | {r['sim']} | {ev} | {r.get('off_limit_count',0)} | {r.get('duration_s',0)} | {r['pass_fail']}{flag} |")
    md.append(f"\n**Mismatches:** {mismatches}")
    md.append(f"\n**M1 ship gate:** {'PASS — all 3 trials match across both sims.' if mismatches == 0 else f'FAIL — {mismatches} mismatch(es). See JSON for details.'}")
    return "\n".join(md) + "\n"


def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--sim", choices=["isaac", "gazebo", "both"], default="both")
    ap.add_argument("--trial", default=None, help="Run only this trial_key (default: all)")
    ap.add_argument("--config", default=DEFAULT_CONFIG)
    args = ap.parse_args()

    cfg = yaml.safe_load(open(args.config))
    trial_keys = sorted(cfg["trials"]) if args.trial is None else [args.trial]

    rows = []
    isaac_outcomes = {}
    gazebo_outcomes = {}

    if args.sim in ("isaac", "both"):
        for tk in trial_keys:
            time_limit = float(cfg["trials"][tk]["tasks"]["task_1"]["time_limit"])
            print(f"[parity_report] Isaac Sim — {tk} (time_limit={time_limit}s)")
            isaac_outcomes[tk] = run_isaac_trial(tk, time_limit)

    if args.sim in ("gazebo", "both"):
        # ONE long Gazebo run captures all trials
        gazebo_outcomes = run_gazebo_all_trials(trial_keys, cfg)

    # Stitch rows + mark mismatches
    for tk in trial_keys:
        i_row = isaac_outcomes.get(tk, {"trial_key": tk, "sim": "isaac", "pass_fail": "SKIPPED"})
        g_row = gazebo_outcomes.get(tk, {"trial_key": tk, "sim": "gazebo", "pass_fail": "SKIPPED"})
        if i_row["pass_fail"] != g_row["pass_fail"] and "SKIPPED" not in (i_row["pass_fail"], g_row["pass_fail"]):
            i_row["_mismatch"] = True; g_row["_mismatch"] = True
        rows.append(i_row)
        rows.append(g_row)

    mismatches = sum(1 for r in rows if r.get("_mismatch")) // 2
    PHASE_DIR.mkdir(parents=True, exist_ok=True)
    (PHASE_DIR / "parity-report.json").write_text(json.dumps(rows, indent=2))
    (PHASE_DIR / "parity-report.md").write_text(render_markdown(rows, mismatches))
    print(f"\n[parity_report] DONE — mismatches={mismatches}")
    print(f"  JSON: {PHASE_DIR / 'parity-report.json'}")
    print(f"  MD:   {PHASE_DIR / 'parity-report.md'}")
    return 0 if mismatches == 0 else 1


if __name__ == "__main__":
    sys.exit(main())
