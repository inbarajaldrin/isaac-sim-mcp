FILE 2 — `entrypoint.sh`:
```bash
#!/bin/bash
# Reference: replaces ~/Documents/aic/docker/aic_eval entrypoint (which runs
# aic_gz_bringup.launch.py wholesale). This entrypoint runs ONLY:
#   - robot_state_publisher (consumes Isaac Sim's URDF state — gives /robot_description)
#   - aic_adapter (the C++ ros2_control plugin's standalone-runnable mode, if available;
#     OR a passthrough that re-publishes /joint_states from Isaac Sim namespace)
#   - aic_engine (trial orchestrator)
#
# Isaac Sim is the source of truth for /joint_states, /tf, /fts_broadcaster/wrench,
# /aic_controller/* — we just need aic_engine to consume those topics.

set -e

# Source the kilted overlay (aic_engine packages are installed here).
source /opt/ros/kilted/setup.bash 2>/dev/null || source /opt/ros/humble/setup.bash
if [ -f /aic_ws/install/setup.bash ]; then
    source /aic_ws/install/setup.bash
fi

# Trial config path defaults to upstream sample_config.yaml inside the image.
CONFIG_PATH="${CONFIG_PATH:-/aic_ws/install/aic_engine/share/aic_engine/config/sample_config.yaml}"
TRIAL_KEY="${TRIAL_KEY:-trial_1}"

echo "[my-eval-isaac] ROS_DOMAIN_ID=$ROS_DOMAIN_ID  RMW=$RMW_IMPLEMENTATION"
echo "[my-eval-isaac] aic_engine config: $CONFIG_PATH  trial: $TRIAL_KEY"

# robot_state_publisher: consume the AIC URDF and re-publish TF.
# (Isaac Sim already publishes /tf for the robot, so this MAY be redundant —
# included for parity with Gazebo's launch graph; can be skipped via SKIP_RSP=1.)
if [ "${SKIP_RSP:-0}" != "1" ]; then
    URDF_PATH="$(ros2 pkg prefix aic_description 2>/dev/null)/share/aic_description/urdf/ur_gz.urdf.xacro"
    if [ -f "$URDF_PATH" ]; then
        ros2 run robot_state_publisher robot_state_publisher \
            --ros-args -p robot_description:="$(xacro "$URDF_PATH")" &
        RSP_PID=$!
    fi
fi

# aic_adapter (if it has a standalone runnable; otherwise skip — aic_engine should
# work with Isaac Sim's direct topics).
if [ "${SKIP_ADAPTER:-0}" != "1" ] && ros2 pkg list 2>/dev/null | grep -q '^aic_adapter$'; then
    ros2 run aic_adapter aic_adapter --ros-args -p use_sim_time:=true &
    ADAPTER_PID=$!
fi

# aic_engine — the actual trial orchestrator.
exec ros2 run aic_engine aic_engine \
    --ros-args \
    -p config_file_path:="$CONFIG_PATH" \
    -p use_sim_time:=true \
    "$@"
```
